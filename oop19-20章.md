```cpp
#include <iostream>
#include <string>

using namespace std;

template<typename T>
class My
{
public:
	void f(const T& x)
	{
		cout << typeid(x).name() << endl;
	}
};

int main()
{
	My<int>obj1;
	My<float>obj2;

	obj1.f(10);//输出int
	obj2.f(20);//输出float
}
```

### 动态多态性

动态多态：

#### C++中的动态多态性-例

```cpp
class B;
class A
{
public:
    virtual ~A( ) { }
    virtual void f(A *)  { cout<<1<<endl;}
    virtual void f(B *)  { cout<<2<<endl; }
};

class B:public A
{
public:
     virtual ~B( ){ }
     virtual void f(A *)  { cout<<3<<endl;}
     virtual void f(B *)  { cout<<4<<endl;}
};
```

```cpp
int main()
{
    B b;
    A*pa = &b; //&b隐式转换为*B
    pa->f(&b);//4
    pa->f(pa);//3
    b.f(pa);//3
    return 0;
}
```

#### 虚机制的作用

虚机制是实现动态多态的一种方法；
意义-在保持客户端访问接口不变的前提下，可以变更类的实现
